132    /* Write with I2C bus */



133    int EXT_I2C_writeH(



134            DEV_entry_t * dev, 



135            const uint8_T address, 



136            const void *data, 



137            const int count)



138    {



139        int ret;



140        struct i2c_rdwr_ioctl_data i2c_data;



141        struct i2c_msg msg;



142



143        if (dev != (DEV_entry_t *)NULL)



144        {



145            // Fill-in I2C message for the register to read



146            msg.addr  = address;



147            msg.flags = 0; 



148            msg.len   = count;



149            msg.buf   = data; 



150            i2c_data.msgs = &msg;



151            i2c_data.nmsgs = 1;



152    



153            /* Write data */



154            ret = ioctl(dev->fd, I2C_RDWR, &i2c_data);



155            if (ret == -1) {



156                perror("EXT_I2C_write/write");



157                return ERR_I2C_WRITE_REGISTER_WRITE;



158            }



159        }



160        else



161        {



162            perror("EXT_I2C_write/write");



163            return ERR_I2C_WRITE_REGISTER_WRITE;



164        }



165



166        return 0;



167    }

169    /* Read with I2C bus */



170    int EXT_I2C_readH(



171            DEV_entry_t *dev, 



172            const uint8_T address, 



173            void *data, 



174            const int count)



175    {



176        int ret;



177        struct i2c_rdwr_ioctl_data i2c_data;



178        struct i2c_msg msg;



179



180        if (dev != (DEV_entry_t *)NULL)



181        {



182            // Fill-in I2C message for the register to read



183            msg.addr  = address;



184            msg.flags = I2C_M_RD;  



185            msg.len   = count;



186            msg.buf   = data; 



187            i2c_data.msgs = &msg;



188            i2c_data.nmsgs = 1;



189    



190            /* Get data from I2C bus */



191            ret = ioctl(dev->fd, I2C_RDWR, &i2c_data);



192            if (ret == -1) {



193                perror("EXT_I2C_read/read");



194                return ERR_I2C_READ_READ;



195            }



196        }



197        else



198        {



199            perror("EXT_I2C_read/read");



200            return ERR_I2C_READ_READ;



201        }



202



203        return 0;



204    }

206    // Initialize I2C device



207    DEV_entry_t * EXT_I2C_initH(const unsigned int bus)



208    {



209        DEV_entry_t *dev = NULL;



210



211        // Get device handle



212        dev = DEV_get(DEV_I2C_0 + bus);



213        if (dev->fd < 0) {



214            dev->fd = I2C_open(bus);



215            if (dev->fd < 0) {



216                perror("I2C_init/I2C_open");



217                /*return ERR_I2C_INIT;*/



218                return (DEV_entry_t *)NULL;



219            }



220        }



221        DEV_lock(dev);



222



223        return dev;



224    }



225



226    // Close I2C device 



227    int EXT_I2C_terminateH(DEV_entry_t * dev) 



228    {



229        if (dev != (DEV_entry_t *)NULL)



230        {



231            DEV_unlock(dev);



232            if (dev->refCount == 0) {



233                if (dev->fd > 0) {



234                    I2C_close(dev->fd);



235                    dev->fd = -1;



236                }



237            }



238        }



239



240        return 0;



241    }
